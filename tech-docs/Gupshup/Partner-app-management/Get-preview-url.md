# Gupshup Get Preview URL API

## Overview
The Get Preview URL API allows you to generate a preview URL for a specific flow, enabling you to test and preview the flow before publishing it. This API provides a temporary URL that expires after a certain time period.

## Base URL
```
https://partner.gupshup.io/partner/app/{appId}/flows/{flowId}/preview
```

## Authentication
- **Method**: Bearer Token
- **Header**: `Authorization: {PARTNER_APP_TOKEN}`

## Endpoint

### Get Preview URL
**GET** `/partner/app/{appId}/flows/{flowId}/preview`

Generates a preview URL for the specified flow.

#### Parameters

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| Authorization | String | Yes | Access Token for the application |
| appId | String | Yes | App ID of the application |
| flowId | String | Yes | Flow ID generated by Meta |

#### Sample cURL Request
```bash
curl --location --request GET 'https://partner.gupshup.io/partner/app/{{APP_ID}}/flows/{{FLOW_ID}}/preview' \
--header 'Authorization: {{PARTNER_APP_TOKEN}}' \
--header 'Content-Type: application/json'
```

**Real Example:**
```bash
curl --location --request GET 'https://partner.gupshup.io/partner/app/YOUR_APP_ID/flows/YOUR_FLOW_ID/preview' \
--header 'Authorization: YOUR_PARTNER_APP_TOKEN' \
--header 'Content-Type: application/json'
```

#### Sample Response
```json
{
    "id": "1616289422261049",
    "preview": {
        "expires_at": "2024-07-21T10:15:47+0000",
        "preview_url": "https://business.facebook.com/wa/manage/flows/1616289422261049/preview/?token=fe2fa8b0-86bf-4d53-a6c3-eec4085ab48b"
    },
    "status": "success"
}
```

#### Status Codes

| Status | Response | Description |
|--------|----------|-------------|
| 200 | Success response with preview URL | Preview URL generated successfully |
| 400 | `{ "status": "error", "message": "Unsupported get request. Object with ID '...' does not exist, cannot be loaded due to missing permissions, or does not support this operation." }` | For invalid flow ID |
| 401 | `{ "status": "error", "message": "Unauthorized" }` | Invalid or missing authorization token |
| 404 | `{ "status": "error", "message": "Flow not found" }` | Flow ID not found |

### Basic Get Preview URL
```javascript
const https = require('https');

async function getPreviewUrl(appId, flowId, partnerToken) {
    const options = {
        hostname: 'partner.gupshup.io',
        port: 443,
        path: `/partner/app/${appId}/flows/${flowId}/preview`,
        method: 'GET',
        headers: {
            'Authorization': partnerToken,
            'Content-Type': 'application/json'
        }
    };

    return new Promise((resolve, reject) => {
        const req = https.request(options, (res) => {
            let responseData = '';
            
            res.on('data', (chunk) => {
                responseData += chunk;
            });
            
            res.on('end', () => {
                try {
                    const parsedData = JSON.parse(responseData);
                    resolve(parsedData);
                } catch (error) {
                    reject(new Error('Invalid JSON response'));
                }
            });
        });

        req.on('error', (error) => {
            reject(error);
        });

        req.end();
    });
}
```

### Advanced Preview URL Manager
```javascript
const fetch = require('node-fetch');

class FlowPreviewManager {
    constructor(baseUrl, partnerToken) {
        this.baseUrl = baseUrl || 'https://partner.gupshup.io';
        this.partnerToken = partnerToken;
        this.previewCache = new Map();
    }

    async getPreviewUrl(appId, flowId) {
        const url = `${this.baseUrl}/partner/app/${appId}/flows/${flowId}/preview`;
        
        try {
            const response = await fetch(url, {
                method: 'GET',
                headers: {
                    'Authorization': this.partnerToken,
                    'Content-Type': 'application/json'
                }
            });

            if (!response.ok) {
                const errorData = await response.json();
                throw new Error(`HTTP error! status: ${response.status}, message: ${errorData.message}`);
            }

            return await response.json();
        } catch (error) {
            throw new Error(`Failed to get preview URL: ${error.message}`);
        }
    }

    async getPreviewUrlWithCache(appId, flowId) {
        const cacheKey = `${appId}-${flowId}`;
        const cached = this.previewCache.get(cacheKey);
        
        // Check if cached URL is still valid
        if (cached && this.isPreviewUrlValid(cached)) {
            return cached;
        }

        const previewData = await this.getPreviewUrl(appId, flowId);
        this.previewCache.set(cacheKey, previewData);
        
        return previewData;
    }

    isPreviewUrlValid(previewData) {
        if (!previewData.preview || !previewData.preview.expires_at) {
            return false;
        }
        
        const expiresAt = new Date(previewData.preview.expires_at);
        const now = new Date();
        
        return expiresAt > now;
    }

    async getPreviewUrlWithRetry(appId, flowId, maxRetries = 3) {
        let lastError;
        
        for (let attempt = 1; attempt <= maxRetries; attempt++) {
            try {
                return await this.getPreviewUrl(appId, flowId);
            } catch (error) {
                lastError = error;
                console.log(`Attempt ${attempt} failed: ${error.message}`);
                
                if (attempt < maxRetries) {
                    const delay = Math.pow(2, attempt) * 1000;
                    await this.sleep(delay);
                }
            }
        }
        
        throw new Error(`Failed to get preview URL after ${maxRetries} attempts: ${lastError.message}`);
    }

    async openPreviewInBrowser(appId, flowId) {
        try {
            const previewData = await this.getPreviewUrlWithCache(appId, flowId);
            const previewUrl = previewData.preview.preview_url;
            
            // Open in browser (Node.js environment)
            const { exec } = require('child_process');
            const platform = process.platform;
            
            let command;
            if (platform === 'win32') {
                command = `start ${previewUrl}`;
            } else if (platform === 'darwin') {
                command = `open ${previewUrl}`;
            } else {
                command = `xdg-open ${previewUrl}`;
            }
            
            exec(command, (error) => {
                if (error) {
                    console.error('Failed to open browser:', error);
                } else {
                    console.log('Preview opened in browser');
                }
            });
            
            return previewData;
        } catch (error) {
            throw new Error(`Failed to open preview: ${error.message}`);
        }
    }

    async validatePreviewUrl(appId, flowId) {
        try {
            const previewData = await this.getPreviewUrl(appId, flowId);
            
            // Check if the preview URL is accessible
            const previewUrl = previewData.preview.preview_url;
            const response = await fetch(previewUrl, { method: 'HEAD' });
            
            return {
                valid: response.ok,
                previewData,
                httpStatus: response.status,
                accessible: response.ok
            };
        } catch (error) {
            return {
                valid: false,
                error: error.message,
                accessible: false
            };
        }
    }

    async getMultiplePreviewUrls(flows) {
        const results = [];
        
        for (const { appId, flowId } of flows) {
            try {
                const previewData = await this.getPreviewUrlWithCache(appId, flowId);
                results.push({
                    appId,
                    flowId,
                    success: true,
                    previewData
                });
            } catch (error) {
                results.push({
                    appId,
                    flowId,
                    success: false,
                    error: error.message
                });
            }
        }
        
        return results;
    }

    clearCache() {
        this.previewCache.clear();
    }

    getCacheStatistics() {
        const stats = {
            totalEntries: this.previewCache.size,
            validEntries: 0,
            expiredEntries: 0
        };
        
        for (const [key, value] of this.previewCache) {
            if (this.isPreviewUrlValid(value)) {
                stats.validEntries++;
            } else {
                stats.expiredEntries++;
            }
        }
        
        return stats;
    }

    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

// Usage
const previewManager = new FlowPreviewManager('https://partner.gupshup.io', 'your-partner-token');

// Get preview URL with caching
previewManager.getPreviewUrlWithCache('your-app-id', 'your-flow-id')
    .then(previewData => {
        console.log('Preview URL:', previewData.preview.preview_url);
        console.log('Expires at:', previewData.preview.expires_at);
    })
    .catch(error => {
        console.error('Error:', error.message);
    });

// Validate preview URL
previewManager.validatePreviewUrl('your-app-id', 'your-flow-id')
    .then(validation => {
        console.log('Validation result:', validation);
    })
    .catch(error => {
        console.error('Validation error:', error);
```

### Enhanced Preview URL Manager with Analytics
```javascript
const axios = require('axios');

class AdvancedPreviewManager {
    constructor(baseUrl, partnerToken) {
        this.baseUrl = baseUrl || 'https://partner.gupshup.io';
        this.partnerToken = partnerToken;
        this.previewCache = new Map();
        this.analytics = {
            totalRequests: 0,
            successfulRequests: 0,
            failedRequests: 0,
            cacheHits: 0,
            cacheMisses: 0
        };
    }

    async getPreviewUrl(appId, flowId) {
        this.analytics.totalRequests++;
        
        try {
            const response = await axios.get(
                `${this.baseUrl}/partner/app/${appId}/flows/${flowId}/preview`,
                {
                    headers: {
                        'Authorization': this.partnerToken,
                        'Content-Type': 'application/json'
                    }
                }
            );

            this.analytics.successfulRequests++;
            return response.data;
        } catch (error) {
            this.analytics.failedRequests++;
            throw new Error(`Failed to get preview URL: ${error.message}`);
        }
    }

    async getPreviewUrlWithAnalytics(appId, flowId) {
        const cacheKey = `${appId}-${flowId}`;
        const cached = this.previewCache.get(cacheKey);
        
        if (cached && this.isPreviewUrlValid(cached)) {
            this.analytics.cacheHits++;
            return cached;
        }

        this.analytics.cacheMisses++;
        const previewData = await this.getPreviewUrl(appId, flowId);
        this.previewCache.set(cacheKey, previewData);
        
        return previewData;
    }

    isPreviewUrlValid(previewData) {
        if (!previewData.preview || !previewData.preview.expires_at) {
            return false;
        }
        
        const expiresAt = new Date(previewData.preview.expires_at);
        const now = new Date();
        const bufferTime = 5 * 60 * 1000; // 5 minutes buffer
        
        return expiresAt > new Date(now.getTime() + bufferTime);
    }

    async generatePreviewReport(appId, flowIds) {
        const report = {
            timestamp: new Date().toISOString(),
            appId,
            totalFlows: flowIds.length,
            successful: 0,
            failed: 0,
            previews: []
        };

        for (const flowId of flowIds) {
            try {
                const previewData = await this.getPreviewUrlWithAnalytics(appId, flowId);
                report.successful++;
                report.previews.push({
                    flowId,
                    success: true,
                    previewUrl: previewData.preview.preview_url,
                    expiresAt: previewData.preview.expires_at,
                    validFor: this.getTimeUntilExpiry(previewData.preview.expires_at)
                });
            } catch (error) {
                report.failed++;
                report.previews.push({
                    flowId,
                    success: false,
                    error: error.message
                });
            }
        }

        return report;
    }

    getTimeUntilExpiry(expiresAt) {
        const now = new Date();
        const expiry = new Date(expiresAt);
        const diff = expiry - now;
        
        if (diff <= 0) {
            return 'Expired';
        }
        
        const hours = Math.floor(diff / (1000 * 60 * 60));
        const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
        
        return `${hours}h ${minutes}m`;
    }

    async refreshExpiredPreviews() {
        const refreshed = [];
        const failed = [];
        
        for (const [key, previewData] of this.previewCache) {
            if (!this.isPreviewUrlValid(previewData)) {
                const [appId, flowId] = key.split('-');
                
                try {
                    const newPreviewData = await this.getPreviewUrl(appId, flowId);
                    this.previewCache.set(key, newPreviewData);
                    refreshed.push({ appId, flowId, success: true });
                } catch (error) {
                    failed.push({ appId, flowId, error: error.message });
                    this.previewCache.delete(key);
                }
            }
        }
        
        return { refreshed, failed };
    }

    async sharePreviewUrl(appId, flowId, method = 'copy') {
        try {
            const previewData = await this.getPreviewUrlWithAnalytics(appId, flowId);
            const previewUrl = previewData.preview.preview_url;
            
            switch (method) {
                case 'copy':
                    // Copy to clipboard (requires additional library in Node.js)
                    console.log('Preview URL (copy to clipboard):', previewUrl);
                    break;
                    
                case 'qr':
                    // Generate QR code (requires qrcode library)
                    const QRCode = require('qrcode');
                    const qrCode = await QRCode.toString(previewUrl, { type: 'terminal' });
                    console.log('QR Code for preview URL:');
                    console.log(qrCode);
                    break;
                    
                case 'email':
                    // Send via email (requires email service)
                    console.log('Email body:', `Preview your flow here: ${previewUrl}`);
                    break;
                    
                default:
                    console.log('Preview URL:', previewUrl);
            }
            
            return previewData;
        } catch (error) {
            throw new Error(`Failed to share preview URL: ${error.message}`);
        }
    }

    async createPreviewUrlBatch(flows, options = {}) {
        const { 
            batchSize = 5, 
            delay = 1000, 
            includeValidation = false 
        } = options;
        
        const results = [];
        
        for (let i = 0; i < flows.length; i += batchSize) {
            const batch = flows.slice(i, i + batchSize);
            const batchResults = await Promise.allSettled(
                batch.map(async ({ appId, flowId }) => {
                    const previewData = await this.getPreviewUrlWithAnalytics(appId, flowId);
                    
                    if (includeValidation) {
                        const validation = await this.validatePreviewUrl(appId, flowId);
                        previewData.validation = validation;
                    }
                    
                    return { appId, flowId, previewData };
                })
            );
            
            batchResults.forEach((result, index) => {
                const { appId, flowId } = batch[index];
                
                if (result.status === 'fulfilled') {
                    results.push({
                        appId,
                        flowId,
                        success: true,
                        data: result.value
                    });
                } else {
                    results.push({
                        appId,
                        flowId,
                        success: false,
                        error: result.reason.message
                    });
                }
            });
            
            // Add delay between batches
            if (i + batchSize < flows.length) {
                await this.sleep(delay);
            }
        }
        
        return results;
    }

    async validatePreviewUrl(appId, flowId) {
        try {
            const previewData = await this.getPreviewUrlWithAnalytics(appId, flowId);
            const previewUrl = previewData.preview.preview_url;
            
            // Check URL accessibility
            const response = await fetch(previewUrl, { method: 'HEAD' });
            
            return {
                valid: true,
                accessible: response.ok,
                httpStatus: response.status,
                expiresAt: previewData.preview.expires_at,
                timeUntilExpiry: this.getTimeUntilExpiry(previewData.preview.expires_at)
            };
        } catch (error) {
            return {
                valid: false,
                accessible: false,
                error: error.message
            };
        }
    }

    getAnalytics() {
        const cacheStats = this.getCacheStatistics();
        
        return {
            requests: this.analytics,
            cache: cacheStats,
            performance: {
                successRate: this.analytics.totalRequests > 0 ? 
                    ((this.analytics.successfulRequests / this.analytics.totalRequests) * 100).toFixed(2) + '%' : 
                    '0%',
                cacheHitRate: (this.analytics.cacheHits + this.analytics.cacheMisses) > 0 ? 
                    ((this.analytics.cacheHits / (this.analytics.cacheHits + this.analytics.cacheMisses)) * 100).toFixed(2) + '%' : 
                    '0%'
            }
        };
    }

    getCacheStatistics() {
        const stats = {
            totalEntries: this.previewCache.size,
            validEntries: 0,
            expiredEntries: 0
        };
        
        for (const [key, value] of this.previewCache) {
            if (this.isPreviewUrlValid(value)) {
                stats.validEntries++;
            } else {
                stats.expiredEntries++;
            }
        }
        
        return stats;
    }

    exportPreviewData(format = 'json') {
        const data = {
            timestamp: new Date().toISOString(),
            analytics: this.getAnalytics(),
            previews: []
        };
        
        for (const [key, previewData] of this.previewCache) {
            const [appId, flowId] = key.split('-');
            data.previews.push({
                appId,
                flowId,
                previewUrl: previewData.preview.preview_url,
                expiresAt: previewData.preview.expires_at,
                valid: this.isPreviewUrlValid(previewData)
            });
        }
        
        if (format === 'csv') {
            return this.convertToCSV(data.previews);
        }
        
        return data;
    }

    convertToCSV(previews) {
        const headers = ['App ID', 'Flow ID', 'Preview URL', 'Expires At', 'Valid'];
        const csvRows = [headers.join(',')];
        
        previews.forEach(preview => {
            const row = [
                preview.appId,
                preview.flowId,
                `"${preview.previewUrl}"`,
                preview.expiresAt,
                preview.valid
            ];
            csvRows.push(row.join(','));
        });
        
        return csvRows.join('\n');
    }

    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}
```

### Preview URL Monitoring and Alerts
```javascript
class PreviewUrlMonitor {
    constructor(previewManager) {
        this.previewManager = previewManager;
        this.monitoringInterval = null;
        this.alerts = [];
        this.thresholds = {
            expiryWarning: 30 * 60 * 1000, // 30 minutes
            failureRate: 0.1 // 10%
        };
    }

    startMonitoring(appId, flowIds, interval = 300000) { // 5 minutes
        this.monitoringInterval = setInterval(async () => {
            await this.checkPreviewUrls(appId, flowIds);
        }, interval);
        
        console.log('Preview URL monitoring started');
    }

    stopMonitoring() {
        if (this.monitoringInterval) {
            clearInterval(this.monitoringInterval);
            this.monitoringInterval = null;
            console.log('Preview URL monitoring stopped');
        }
    }

    async checkPreviewUrls(appId, flowIds) {
        const results = [];
        
        for (const flowId of flowIds) {
            try {
                const previewData = await this.previewManager.getPreviewUrlWithAnalytics(appId, flowId);
                const validation = await this.previewManager.validatePreviewUrl(appId, flowId);
                
                results.push({
                    flowId,
                    success: true,
                    previewData,
                    validation
                });
                
                // Check for expiry warnings
                await this.checkExpiryWarning(appId, flowId, previewData);
                
            } catch (error) {
                results.push({
                    flowId,
                    success: false,
                    error: error.message
                });
                
                await this.createAlert('PREVIEW_FAILURE', `Failed to get preview for flow ${flowId}: ${error.message}`);
            }
        }
        
        // Check overall failure rate
        const failureRate = results.filter(r => !r.success).length / results.length;
        if (failureRate > this.thresholds.failureRate) {
            await this.createAlert('HIGH_FAILURE_RATE', `Preview failure rate is ${(failureRate * 100).toFixed(1)}%`);
        }
        
        return results;
    }

    async checkExpiryWarning(appId, flowId, previewData) {
        const expiresAt = new Date(previewData.preview.expires_at);
        const now = new Date();
        const timeToExpiry = expiresAt - now;
        
        if (timeToExpiry <= this.thresholds.expiryWarning && timeToExpiry > 0) {
            await this.createAlert('EXPIRY_WARNING', `Preview URL for flow ${flowId} expires in ${Math.floor(timeToExpiry / 60000)} minutes`);
        }
    }

    async createAlert(type, message) {
        const alert = {
            type,
            message,
            timestamp: new Date().toISOString(),
            id: Date.now().toString()
        };
        
        this.alerts.push(alert);
        
        // Send notification
        await this.sendNotification(alert);
        
        return alert;
    }

    async sendNotification(alert) {
        // In production, send to monitoring service, email, Slack, etc.
        console.log(`🚨 ALERT [${alert.type}]: ${alert.message}`);
    }

    getAlerts(sinceTimestamp = null) {
        if (sinceTimestamp) {
            return this.alerts.filter(alert => new Date(alert.timestamp) > new Date(sinceTimestamp));
        }
        return this.alerts;
    }

    clearAlerts() {
        this.alerts = [];
    }

    setThresholds(thresholds) {
        this.thresholds = { ...this.thresholds, ...thresholds };
    }

    getMonitoringStatus() {
        return {
            active: this.monitoringInterval !== null,
            alertCount: this.alerts.length,
            thresholds: this.thresholds,
            lastCheck: new Date().toISOString()
        };
    }
}

// Usage
const previewManager = new AdvancedPreviewManager('https://partner.gupshup.io', 'your-partner-token');
const monitor = new PreviewUrlMonitor(previewManager);

// Start monitoring
const flowIds = ['flow1', 'flow2', 'flow3'];
monitor.startMonitoring('your-app-id', flowIds, 300000); // Check every 5 minutes

// Set custom thresholds
monitor.setThresholds({
    expiryWarning: 60 * 60 * 1000, // 1 hour
    failureRate: 0.05 // 5%
});

// Check monitoring status
console.log('Monitoring Status:', monitor.getMonitoringStatus());
```

```javascript
class CompletePreviewManager {
    constructor(config) {
        this.baseUrl = config.baseUrl || 'https://partner.gupshup.io';
        this.partnerToken = config.partnerToken;
        this.cache = new Map();
        this.analytics = new Map();
        this.monitor = null;
        this.config = config;
    }

    async getPreviewUrl(appId, flowId, options = {}) {
        const { 
            useCache = true, 
            validateUrl = false, 
            openInBrowser = false 
        } = options;
        
        try {
            let previewData;
            
            if (useCache) {
                previewData = await this.getPreviewUrlWithCache(appId, flowId);
            } else {
                previewData = await this.fetchPreviewUrl(appId, flowId);
            }
            
            if (validateUrl) {
                const validation = await this.validatePreviewUrl(appId, flowId);
                previewData.validation = validation;
            }
            
            if (openInBrowser) {
                await this.openInBrowser(previewData.preview.preview_url);
            }
            
            this.recordAnalytics(appId, flowId, 'success');
            return previewData;
            
        } catch (error) {
            this.recordAnalytics(appId, flowId, 'error', error.message);
            throw error;
        }
    }

    async fetchPreviewUrl(appId, flowId) {
        const response = await fetch(`${this.baseUrl}/partner/app/${appId}/flows/${flowId}/preview`, {
            method: 'GET',
            headers: {
                'Authorization': this.partnerToken,
                'Content-Type': 'application/json'
            }
        });

        if (!response.ok) {
            const errorData = await response.json();
            throw new Error(`HTTP error! status: ${response.status}, message: ${errorData.message}`);
        }

        return await response.json();
    }

    async getPreviewUrlWithCache(appId, flowId) {
        const cacheKey = `${appId}-${flowId}`;
        const cached = this.cache.get(cacheKey);
        
        if (cached && this.isValidPreview(cached)) {
            return cached;
        }

    }

    isValidPreview(previewData) {
        if (!previewData.preview || !previewData.preview.expires_at) {
            return false;
        }
        
        const expiresAt = new Date(previewData.preview.expires_at);
        const now = new Date();
        const buffer = 5 * 60 * 1000; // 5 minutes buffer
        
        return expiresAt > new Date(now.getTime() + buffer);
    }

    async validatePreviewUrl(appId, flowId) {
        try {
            const previewData = await this.getPreviewUrlWithCache(appId, flowId);
            const previewUrl = previewData.preview.preview_url;
            
            const response = await fetch(previewUrl, { method: 'HEAD' });
            
            return {
                valid: true,
                accessible: response.ok,
                httpStatus: response.status,
                expiresAt: previewData.preview.expires_at
            };
        } catch (error) {
            return {
                valid: false,
                accessible: false,
                error: error.message
            };
        }
    }

    async openInBrowser(previewUrl) {
        const { exec } = require('child_process');
        const platform = process.platform;
        
        let command;
        if (platform === 'win32') {
            command = `start ${previewUrl}`;
        } else if (platform === 'darwin') {
            command = `open ${previewUrl}`;
        } else {
            command = `xdg-open ${previewUrl}`;
        }
        
        return new Promise((resolve) => {
            exec(command, (error) => {
                if (error) {
                    console.error('Failed to open browser:', error);
                    resolve(false);
                } else {
                    console.log('Preview opened in browser');
                    resolve(true);
                }
            });
        });
    }

    recordAnalytics(appId, flowId, status, error = null) {
        const key = `${appId}-${flowId}`;
        const analytics = this.analytics.get(key) || {
            totalRequests: 0,
            successfulRequests: 0,
            failedRequests: 0,
            errors: []
        };
        
        analytics.totalRequests++;
        
        if (status === 'success') {
            analytics.successfulRequests++;
        } else {
            analytics.failedRequests++;
            if (error) {
                analytics.errors.push({
                    timestamp: new Date().toISOString(),
                    error
                });
            }
        }
        
        this.analytics.set(key, analytics);
    }

    getAnalytics(appId = null, flowId = null) {
        if (appId && flowId) {
            return this.analytics.get(`${appId}-${flowId}`);
        }
        
        const allAnalytics = {};
        for (const [key, analytics] of this.analytics) {
            allAnalytics[key] = analytics;
        }
        
        return allAnalytics;
    }

    async batchGetPreviewUrls(flows, options = {}) {
        const { batchSize = 5, delay = 1000 } = options;
        const results = [];
        
        for (let i = 0; i < flows.length; i += batchSize) {
            const batch = flows.slice(i, i + batchSize);
            const batchPromises = batch.map(({ appId, flowId }) => 
                this.getPreviewUrl(appId, flowId, options)
                    .then(previewData => ({ appId, flowId, success: true, previewData }))
                    .catch(error => ({ appId, flowId, success: false, error: error.message }))
            );
            
            const batchResults = await Promise.all(batchPromises);
            results.push(...batchResults);
            
            if (i + batchSize < flows.length) {
                await this.sleep(delay);
            }
        }
        
        return results;
    }

    async exportData(format = 'json') {
        const data = {
            timestamp: new Date().toISOString(),
            cache: this.getCacheData(),
            analytics: this.getAnalytics()
        };
        
        if (format === 'csv') {
            return this.convertToCSV(data);
        }
        
        return data;
    }

    getCacheData() {
        const cacheData = [];
        
        for (const [key, previewData] of this.cache) {
            const [appId, flowId] = key.split('-');
            cacheData.push({
                appId,
                flowId,
                previewUrl: previewData.preview.preview_url,
                expiresAt: previewData.preview.expires_at,
                valid: this.isValidPreview(previewData)
            });
        }
        
        return cacheData;
    }

    convertToCSV(data) {
        const headers = ['App ID', 'Flow ID', 'Preview URL', 'Expires At', 'Valid', 'Success Rate'];
        const csvRows = [headers.join(',')];
        
        data.cache.forEach(item => {
            const analytics = this.getAnalytics(item.appId, item.flowId) || {};
            const successRate = analytics.totalRequests > 0 ? 
                ((analytics.successfulRequests / analytics.totalRequests) * 100).toFixed(2) + '%' : 
                'N/A';
            
            const row = [
                item.appId,
                item.flowId,
                `"${item.previewUrl}"`,
                item.expiresAt,
                item.valid,
                successRate
            ];
            csvRows.push(row.join(','));
        });
        
        return csvRows.join('\n');
    }

    clearCache() {
        this.cache.clear();
    }

    clearAnalytics() {
        this.analytics.clear();
    }

    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}
```

## Error Handling

```javascript
function handlePreviewUrlError(error) {
    if (error.response) {
        const { status, data } = error.response;
        
        switch (status) {
            case 400:
                console.error('Bad Request: Invalid flow ID or flow does not exist');
                return 'INVALID_FLOW_ID';
            case 401:
                console.error('Unauthorized: Invalid partner token');
                return 'UNAUTHORIZED';
            case 403:
                console.error('Forbidden: Insufficient permissions');
                return 'FORBIDDEN';
            case 404:
                console.error('Not Found: Flow not found');
                return 'NOT_FOUND';
            case 500:
                console.error('Server Error: Try again later');
                return 'SERVER_ERROR';
            default:
                console.error('Unknown error:', data);
                return 'UNKNOWN_ERROR';
        }
    } else if (error.request) {
        console.error('Network error: No response received');
        return 'NETWORK_ERROR';
    } else {
        console.error('Error:', error.message);
        return 'UNKNOWN_ERROR';
    }
}
```

## Best Practices

3. **Error Handling**: Implement comprehensive error handling
4. **Validation**: Validate preview URLs before sharing
5. **Monitoring**: Monitor preview URL generation and expiry
6. **Batch Operations**: Process multiple preview URLs efficiently
7. **Analytics**: Track preview URL usage and success rates
8. **Security**: Never expose preview URLs in client-side code
9. **Rate Limiting**: Implement rate limiting for batch operations
10. **Cleanup**: Regularly clean up expired cache entries
